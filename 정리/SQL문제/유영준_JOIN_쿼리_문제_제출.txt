1. 사원명, 부서번호, 부서이름, 부서 근무지를 조회  
select 
e.ename,
e.DEPTNO,
d.dname,
d.loc
from emp as e 
join dept as d 
on e.DEPTNO = d.DEPTNO

2. 부서별 급여합계와 부서이름을 조회  
select 
sum(e.SAL),
d.dname
from emp as e 
join dept as d 
on e.DEPTNO = d.DEPTNO 
group by dname

3. 부서 근무지가 NEW YORK이고, 직책이 MANAGER인 사원의 이름과 급여를 조회  
select 
e.ename,
e.SAL
from emp as e 
join dept as d 
on e.DEPTNO = d.DEPTNO 
where d.loc = 'new york'
and e.job = 'manager'

4. 1983년 이후 입사한 사원의 보너스가 null이면 100 주고, 사원의 이름과 부서이름, 직업을 조회  
select 
ifnull(COMM,100),
e.ename,
d.dname,
e.job
from emp as e 
join dept as d 
on e.DEPTNO = d.DEPTNO 
where e.HIREDATE >= date_format(830101,'%y-%m-%d')
