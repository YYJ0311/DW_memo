- +=, ++ 연산자

		레벨 = 레벨 +1;

		// 사칙연산 *, /, %, +, - 이용

		// 레벨 *= 2 의 의미 : 레벨 = 레벨 * 2, 나머지 연산도 동일
		++레벨; 		// 증감연산자, 레벨을 1만 더함
					// 매우 많이 씀 !!
					// 레벨++ 의 사용은 나중에 배움
		System.out.println("내가 선택한 챔프는 : " +캐릭터);
